UDP - 비연결형 Server             Client socket()           socket() bind() recvfrom()         sendto() sendto()           recvfrom() closesocket()      closesocket()===========================================sendto - UDP 통신에서 사용되는 데이터 전송 함수int sendto( SOCKET s, const char * buf, int len, int flags, const struct sockaddr * to, int tolen );- SOCKET s - 연결할 소켓- const char * buf - 버퍼의 주소- int len - 버퍼의 크기- int flags - 0- const struct sockaddr * to - 수신자의 소켓 주소- int tolen - 소켓의 크기recvfrom 함수 - UDP 통신에서 사용되는 데이터 수신 함수int recvfrom( SOCKET s, char * buf, int len, int flags, struct sockaddr * from, int * fromlen );- SOCKET s - 연결할 소켓- const char * buf - 버퍼의 주소- int len - 버퍼의 크기- int flags - 0- struct sockaddr * from - 송신자의 소켓 주소- int * fromlen - 송신자의 소켓 크기를 저장한 변수의 주소값..
